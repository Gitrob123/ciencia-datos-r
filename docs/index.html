set.seed(123)

train_index <- sample(1:nrow(titanic), 0.7*nrow(titanic))
train <- titanic[train_index, ]
test  <- titanic[-train_index, ]

rf_model <- randomForest(
  Survived ~ Class + Sex + Age,
  data = train,
  ntree = 200,
  importance = TRUE
)

1.4 📈 4. Importancia de Variables

importance_df <- as.data.frame(importance(rf_model))
importance_df$Variable <- rownames(importance_df)

plot_ly(
  importance_df,
  x = ~MeanDecreaseGini,
  y = ~reorder(Variable, MeanDecreaseGini),
  type = "bar",
  orientation = "h"
) %>%
  layout(
    title = "Importancia de Variables",
    xaxis = list(title = "Mean Decrease Gini"),

1.5 🎯 5. Evaluación del Modelo

pred <- predict(rf_model, newdata = test)

                     Real = test$Survived)

datatable(as.data.frame(conf_matrix),
          caption = "Matriz de Confusión")
